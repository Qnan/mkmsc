		\frame[label=frreduction]
		{
			\frametitle{Задача редукции полинома}

			\begin{itemize}
				\item Редукция -- это поиск остатка от деления полинома на набор полиномов
				\item Применяется для
						\begin{itemize}
							\item проверки принадлежности идеалу
							\item построения базиса Гр\"ебнера
						\end{itemize}
				\item Простейший алгоритм -- последовательное исключение старшего монома
				\item Сложность существенно зависит от входных данных 
				\item Количество членов в полиноме может интенсивно расти
			\end{itemize}
		}

		\frame[label=frrsvpdef]
		{
			\frametitle{Метод РСВП}
			$$N(\lambda p + \mu q)=\lambda N(p) + \mu N(q)$$
			$$N(p\cdot q)=N(N(p)\cdot N(q))$$
		  \begin{itemize}
				\item Вычисляет нормальную форму снизу вверх
				\item Сохраняет нормальные формы мономов
				\item Нормальная форма каждого монома вычисляется лишь единожды
				\item Сокращается число промежуточных членов
			\end{itemize}
		}
