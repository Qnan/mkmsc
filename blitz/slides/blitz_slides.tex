					\alpha_4 &= y^2		\\
					\beta_1 &= \alpha_3 \cdot \alpha_4 = y^3				\\
					\beta_2 &= \alpha_1 + \alpha_4 = x + y^2 				\\
					\beta_3 &= \beta_2 \cdot \alpha_2 = x^3+x^2y^2 	\\
					\beta_4 &= \beta_3 + \beta_1 = x^3+x^2y^2+y^3

		\frame[label=frrsvpdef]
		{
			\frametitle{Метод РСВП}
			$$N(\lambda p + \mu q)=\lambda N(p) + \mu N(q)$$
			$$N(p\cdot q)=N(N(p)\cdot N(q))$$
		  \begin{itemize}
				\item<1-> Вычисляет нормальную форму снизу вверх
				\item<2-> Сохраняет нормальные формы мономов
				\item<3-> Нормальная форма каждого монома вычисляется лишь единожды
				\item<4-> Сокращается число промежуточных членов
			\end{itemize}
		}

\section[Выводы]{Concl}
		\frame[label=frconcl]
		{
			\frametitle{Выводы}
			Метод РСВП 
			\begin{itemize}
				\item<1-> обладает высокой гибкостью и эффективностью,
				\item<2-> допускает частично параллельную реализацию,
				\item<3-> позволяет редуцировать несколько полиномов совместно.
			\end{itemize}
			
			\onslide<4-> Однако,	
			\begin{itemize}
				\item<4-> равномерный критерий сложности не отвечает фактической сложности,
				\item<5-> схема, построенная заранее, может быть сколь угодно далека от оптимальной,
				\item<6-> направление подъема существенно влияет на эффективность алгоритма.
			\end{itemize}
		}

		\frame[label=frsolutions]
		{
			\frametitle{Дальше}
			
			Для повышения эффективности данного подхода необходимо
			\begin{itemize}
				\item<1-> строить схему в процессе редукции,
				\item<2-> тщательно выбирать направление подъема,
				\item<3-> добавлять промежуточные мономы лишь если они мало отклоняются от заданного направления.
			\end{itemize}
			
			\onslide<4-> Также необходимо исследовать эффективность данного алгоритма в при работе с булевыми полиномами и другими полиномами высокой размерности.
		}
